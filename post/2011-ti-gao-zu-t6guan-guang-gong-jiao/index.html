不用加速器

题面写了公交车每次都必须等待所有景点上车才能前往下一个景点
所以想到什么(啥都没想到):计算每一站的到达时间 (a记录乘客的信息)

for(int i=1;i<=m;i++)  las[a[i].st]=max(las[a[i].st],a[i].t);

第一站到达时间肯定是las[i]:

arr[1]=las[1]

随后就是一个动态更新的过程，每一站的到达时间就是前一站的发车时间加这一段路的时间

for(int i=2;i<=n;i++) arr[i]=max(arr[i-1],las[i-1])+di[i-1];

最后就是ans了，显然易见，每个乘客的旅游时间=到达时间-出发时间

for(int i=1;i<=m;i++) ans+=arr[a[i].ed]-a[i].t;

好了,10%的分拿上了

使用加速器

这是就要想使用加速器怎样获得最大收益: 废话
简单推理一下:

  • 肯定收益的人更多使用加速器更值钱
  • 但不一定用在人数最多的点上就是最优
  • 可能加速器能影响许多站点(比如人等车的情况,会缩短等待时间)
    • 结论:算出加速器在一个点使用能影响到的最远点，计算两点间的乘客数，比较即可选出最优
      所以啰嗦半天就是要使收益人最多

倒着推方便计算(好像只能倒着推)

        lst[n]=lst[n-1]=n;//预处理
        for(int i=n-2;i>=1;i--)
        {
            if(arr[i+1]<=las[i+1]) 
            {
                //如果到达时间小于最后一个人等待时间，那就是人等车，可以延续
                lst[i]=i+1;
            }
            else lst[i]=lst[i+1];//否则车等人，开再多也没用，就停滞了
        }

题面又说了 每个加速器加速1分钟，所以扫一遍找出缩短时间的最大值

        for(int i=1;i<n;i++)
        {
            int val=pre[lst[i]]-pre[i];//这里用了前缀和，第i点不受益，不是i-1
            if(val>mx&&di[i]>0) //保证Di>=0,所以减1就是保证Di>0
            {
                tar=i;//tar记录的是对那一条路使用加速器，更新要用
                mx=val;
            }
        }

好了下面就是更新了，要更新的有

  1. 答案ans
  2. 使用加速器那条路的通过实践
  3. 到达时间
        ans-=mx;
        di[tar]-=1;
        for(int i=2;i<=n;i++) arr[i]=max(arr[i-1],las[i-1])+di[i-1];

最后输出ans，结束

Code

#include<bits/stdc++.h>
using namespace std;
const int mxn=1e5+10;
template<class T>void read(T &x);

int n,m,k;
int di[mxn];
struct pass{
    int t,a,b;
}a[mxn];
int las[mxn],arr[mxn];
int pre[mxn];
int ans;
int lst[mxn];

int main()
{
    //freopen("bus.in","r",stdin);
    //freopen("bus.out","w",stdout);
    read(n); read(m); read(k);
    for(int i=1;i<n;i++) read(di[i]);
    for(int i=1;i<=m;i++)
    {
        read(a[i].t); read(a[i].a); read(a[i].b);
    }
    for(int i=1;i<=m;i++)
    {
        las[a[i].a]=max(las[a[i].a],a[i].t);
        pre[a[i].b]++;
    }
    arr[1]=las[1];
    for(int i=1;i<=n;i++) pre[i]+=pre[i-1];
    for(int i=2;i<=n;i++)
    {
        arr[i]=max(arr[i-1],las[i-1])+di[i-1];
    }
    for(int i=1;i<=m;i++)
    {
        ans+=arr[a[i].b]-a[i].t;
    }
    while(k--)
    {
        int mx=-1;
        int tar=0;
        lst[n]=lst[n-1]=n;
        for(int i=n-2;i>=1;i--)
        {
            if(arr[i+1]<=las[i+1])
            {
                lst[i]=i+1;
            }
            else lst[i]=lst[i+1];
        }
        for(int i=1;i<n;i++)
        {
            int val=pre[lst[i]]-pre[i];
            if(val>mx&&di[i]>0)
            {
                mx=val;
                tar=i;
            }
        }
        ans-=mx;
        di[tar]-=1;
        for(int i=2;i<=n;i++) arr[i]=max(arr[i-1],las[i-1])+di[i-1];
    }
    printf("%d\n",ans);
    //fclose(stdin); fclose(stdout);
    return 0;
}

template<class T>void read(T &x)
{
	x=0;int f=0;char ch=getchar();
	while(ch<'0'||ch>'9') { f|=(ch=='-'); ch=getchar(); }
	while(ch<='9'&&ch>='0') { x=(x<<1)+(x<<3)+(ch^48); ch=getchar(); }
	x = f ? -x : x;
	return ;
}
