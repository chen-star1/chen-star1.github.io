2011提高组 T3:Mayan游戏

2020-07-01 7 min read # 题解

洛谷传送门

方法：DFS+剪枝

特别简单，写就是了!

定义的变量

const int mxn=10;
int ma[mxn][mxn]; //存储棋盘
int n; //限制步数
int last[mxn][mxn][mxn]; //用于回溯，last[step][i][j]表示第step步时的ma[i][j]
int ans[mxn][mxn]; //答案，包含方块坐标及移动方向
bool de[mxn][mxn]; //临时删除存储，讲删除时会详细讲解

核心

1. check(检查是否消除完毕):

废话，直接全图检查就是了
如果全部消除完，那么最后一行肯定都被消除，只需要枚举最后一行即可

bool check()
{
    for(int i=1;i<=5;i++)  if(ma[i][1]!=0) return false;
    return true;
}

2. update(进行掉落更新):

定义一个hi为下方空块的数量，模拟即可

void update()
{
    int hi;
    for(int i=1;i<=5;i++)
    {
        hi=0;
        for(int j=1;j<=7;j++)
        {
            if(ma[i][j]==0) hi++;
            else 
            {
                swap(ma[i][j],ma[i][j-hi]);
            }
        }
    }
}

3.del(删除)

特别注意:不是见到三个连续的就消除
例如：

如果直接删除下面三个，就会WA
所以del数组就派上用场了，我们先寻找所有能消除的部分，最后一并删除
上图中，中间点其实被记录了两次(一次横，一次竖)

bool del()
{
    bool flag=false;
    for(int i=1;i<=5;i++)
    {
        for(int j=1;j<=7;j++)
        {
            if(i>1&&i<=5&&ma[i][j]&&ma[i][j]==ma[i+1][j]&&ma[i][j]==ma[i-1][j])
            {
                de[i][j]=de[i+1][j]=de[i-1][j]=1;
                flag=true;
            }
            if(j>1&&j<7&&ma[i][j]&&ma[i][j]==ma[i][j+1]&&ma[i][j]==ma[i][j-1])
            {
                de[i][j]=de[i][j+1]=de[i][j-1]=1;
                flag=true;
            }
        }
    }
    if(!flag) return false;
    for(int i=1;i<=5;i++)
    {
        for(int j=1;j<=7;j++)
        {
            if(de[i][j])
            {
                de[i][j]=0;
                ma[i][j]=0;
            }
        }
    }
    return true;
}

定义成bool类型，是因为消除之后可能会再次有方块掉落，掉落后又会产生新的能消除的方块
所以在每次update更新后都要检查能否继续消除

剪枝技巧

1.相同颜色交换可以跳过(必须的)
2.如果向左边移动，那么被交换的位置一定是空块
Why? 显而易见
想一想，如果左边不是空块，那么该步骤就可以变成一个点向右移
而答案要求字典序输出，1优先于-1

注意事项
1.剪枝(上面有了)
2.记得回溯，用last数组调回map(因为有多组解，要寻找最优解更新)
3.答案输出:以(0,0)为起始点，如果以(1,1)为起始点要减一

#include<bits/stdc++.h>
using namespace std;
template<class T>void read(T &x);
const int mxn=10;
int ma[mxn][mxn];
int n;
int last[mxn][mxn][mxn];
int ans[mxn][mxn];
bool de[mxn][mxn];

bool check()
{
    for(int i=1;i<=5;i++)
    {
        if(ma[i][1]!=0) return false;
    }
    return true;
}

void update()
{
    int hi;
    for(int i=1;i<=5;i++)
    {
        hi=0;
        for(int j=1;j<=7;j++)
        {
            if(ma[i][j]==0) hi++;
            else 
            {
                swap(ma[i][j],ma[i][j-hi]);
            }
        }
    }
}

bool del()
{
    bool flag=false;
    for(int i=1;i<=5;i++)
    {
        for(int j=1;j<=7;j++)
        {
            if(i>1&&i<=5&&ma[i][j]&&ma[i][j]==ma[i+1][j]&&ma[i][j]==ma[i-1][j])
            {
                de[i][j]=de[i+1][j]=de[i-1][j]=1;
                flag=true;
            }
            if(j>1&&j<7&&ma[i][j]&&ma[i][j]==ma[i][j+1]&&ma[i][j]==ma[i][j-1])
            {
                de[i][j]=de[i][j+1]=de[i][j-1]=1;
                flag=true;
            }
        }
    }
    if(!flag) return false;
    for(int i=1;i<=5;i++)
    {
        for(int j=1;j<=7;j++)
        {
            if(de[i][j])
            {
                de[i][j]=0;
                ma[i][j]=0;
            }
        }
    }
    return true;
}

void copy(int step)
{
    for(int i=1;i<=5;i++)
    {
        for(int j=1;j<=7;j++)
        {
            ma[i][j]=last[step][i][j];
        }
    }
}

void dfs(int step)
{
    if(check())
    {
        for(int i=1;i<=n;i++)
        {
            printf("%d %d %d\n",ans[i][1],ans[i][2],ans[i][3]);
        }
        exit(0);
    }
    if(step>n) return;

    for(int i=1;i<=5;i++)
    {
        for(int j=1;j<=7;j++) last[step][i][j]=ma[i][j];
    }

    for(int i=1;i<=5;i++)
    {
        for(int j=1;j<=7;j++)
        {
            if(ma[i][j])
            {
                if(i<5&&ma[i][j]!=ma[i+1][j])
                {
                    swap(ma[i][j],ma[i+1][j]);
                    ans[step][1]=i-1; ans[step][2]=j-1; ans[step][3]=1;
                    //do while实现更新并删除
                    do{
                        update();
                    }while(del());
                    dfs(step+1);
                    copy(step);
                }
                if(i>1&&ma[i-1][j]==0)
                {
                    swap(ma[i][j],ma[i-1][j]);
                    ans[step][1]=i-1; ans[step][2]=j-1; ans[step][3]=-1;
                    //do while实现更新并删除
                    do{
                        update();
                    }while(del());
                    dfs(step+1);
                    copy(step);
                }
            }
        }
    }
}

int main()
{
    read(n);
    int in;
    for(int i=1;i<=5;i++)
    {
        for(int j=1;j<=8;j++)
        {
            read(in);
            if(in==0) break;
            ma[i][j]=in;
        }
    }
    dfs(1);
    //如果有解在dfs中直接exit(0),也可以根据ans判断是否有解
    puts("-1");
    return 0;
}

template<class T>void read(T &x)
{
	x=0;int f=0;char ch=getchar();
	while(ch<'0'||ch>'9') { f|=(ch=='-'); ch=getchar(); }
	while(ch<='9'&&ch>='0') { x=(x<<1)+(x<<3)+(ch^48); ch=getchar(); }
	x = f ? -x : x;
	return ;
}
